#ifndef SOUND_H
#define SOUND_H

#include <iostream>

using namespace std;

struct Mix_Chunk;
typedef struct _Mix_Music Mix_Music;

class Sound{
 private:
  Mix_Chunk *w;
  Mix_Music *m;
  bool isMusic;
 public:
  Sound(){}
  Sound(const char* loc, bool music);
  int Setup(const char* loc, bool music);

  //Control
  void Play();
  void Pause();
  void Resume();
  void Stop();
  void SetChunkVolume(float value);

  //Destructor
  ~Sound();
};

#endif
